
CREATE TABLE `administrador` (
  `usernameAdmin` varchar(25) NOT NULL,
  `password` varchar(12) DEFAULT NULL,
  PRIMARY KEY (`usernameAdmin`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8

CREATE TABLE `inventario` (
  `idProducto` int(11) NOT NULL,
  `Cantidad` int(11) DEFAULT NULL,
  PRIMARY KEY (`idProducto`),
  CONSTRAINT `idProducto` FOREIGN KEY (`idProducto`) REFERENCES `producto` (`idProducto`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8

CREATE TABLE `producto` (
  `idProducto` int(11) NOT NULL,
  `nombreProducto` varchar(100) DEFAULT NULL,
  `precio` int(11) DEFAULT NULL,
  `detalle` varchar(500) DEFAULT NULL,
  PRIMARY KEY (`idProducto`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8

CREATE TABLE `usuario` (
  `username` varchar(25) NOT NULL,
  `password` varchar(12) DEFAULT NULL,
  PRIMARY KEY (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8

CREATE TABLE `proyecto`.`orden` (
  `idOrden` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(25) NOT NULL,
  PRIMARY KEY (`idOrden`),
  INDEX `username_idx` (`username` ASC),
  CONSTRAINT `username`
    FOREIGN KEY (`username`)
    REFERENCES `proyecto`.`usuario` (`username`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);


CREATE TABLE `proyecto`.`productoxorden` (
  `idOrden` INT NOT NULL,
  `idProducto` INT NOT NULL,
  `cantidad` INT NOT NULL,
  INDEX `idOrden_idx` (`idOrden` ASC),
  CONSTRAINT `idOrden`
    FOREIGN KEY (`idOrden`)
    REFERENCES `proyecto`.`orden` (`idOrden`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
